import { useEffect, useState } from "react";

function App() {
  const [msg, setMsg] = useState("");

  useEffect(() => {
    fetch("/api")
      .then(res => res.json())
      .then(data => setMsg(data.message));
  }, []);

  return (
    <div style={{ padding: "40px", fontSize: "20px" }}>
      <h1>AWSWebHub</h1>
      <p>{msg}</p>
    </div>
  );
}

export default App;
